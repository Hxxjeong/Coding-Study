# [소수](https://www.acmicpc.net/problem/2581)

<br>

## 시간복잡도 분석
-  O(1)

## 코드
```java
package baekjoon.bronze;

import java.io.*;

public class b251013_2581_소수 {    // 완전탐색
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int M = Integer.parseInt(br.readLine());
        int N = Integer.parseInt(br.readLine());

        int sum = 0; // 소수의 합
        // 아주 큰 수로 초기화 <- 처음 min 보다 작아서 무조건 갱신.
        int min = Integer.MAX_VALUE; 

        // M부터 N까지 반복하면서 소수 찾기 - 완전탐색
        for (int i = M; i <= N; i++) {
            if (isPrime(i)) {
                sum += i;
                if (i < min) {
                    min = i; // 더 작은 소수가 나올 때마다 업데이트
                }
            }
        }

        if (sum == 0) {
            System.out.println(-1);
        } else {
            System.out.println(sum); 
            System.out.println(min); 
        }
    }

    // 소수 판별 함수
    public static boolean isPrime(int num) {
        if (num < 2) return false;   // 1 이하는 소수 아님
        
        // 소수 판별: 2부터 √i 까지만 확인
        // sqrt : 제곱근(√num) 을 계산해주는 함수
        // num이 소수인지 판별할 때, 
        // 약수찾을 때 제곱근까지만 검사해도 소수인지 아닌지 판별이 가능함
        for (int i = 2; i <= Math.sqrt(num); i++) {
            // 나누었을 때 0이 되면 나누어 떨어진 것이므로 소수가 아님
            if (num % i == 0) return false;    
        }

        return true;
    }
}

```
